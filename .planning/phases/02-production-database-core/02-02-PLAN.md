```typescript
import mustache from 'mustache';
import { getREDNodes } from '../lib/red';
import type { PostgresNodeConfig } from '../lib/types';

export function PostgresNode(this: any, config: PostgresNodeConfig) {
  const RED = getREDNodes();
  const node = this;
  RED.createNode(node, config);
  node.topic = config.topic;
  node.config = RED.getNode(config.PostgresDBNode);

  node.on('input', (msg: any) => {
    const query = mustache.render(config.query, { msg });
    const asyncQuery = async () => {
      let client = null;
      try {
        client = await node.config.pgPool.connect();
        msg.payload = await client.query(query, msg.params || []);
        node.status({ fill: 'green', shape: 'ring', text: `Query ok. ${msg.payload.rowCount} rows returned` });
      } catch (err: any) {
        const errorMessage = `Error executing query: ${err.message}`;
        node.status({ fill: 'red', shape: 'ring', text: errorMessage });
        if (config.throwErrors) { node.error(errorMessage, msg); msg = null; }
        else { node.error(errorMessage); msg.error = errorMessage; }
      } finally {
        if (client) { try { client.release(); } catch(releaseErr) { ... } }
        node.send(msg);
      }
    };
    asyncQuery().catch((unhandledError) => { node.error(`Unhandled error: ${unhandledError.message}`); });
  });
}
```

## the agent's Discretion (resolved in this plan)

- **Named parameters mapping (QUERY-01):** Extract `$N` references from SQL via regex, sort numerically, then map `msg.params` keys by their object insertion order (as returned by `Object.keys()`). Document that `msg.params` keys should be ordered to match `$1, $2, $3`. Mustache rendering runs BEFORE parameter binding. When `msg.params` is an array (existing behavior), pass through unchanged — backward compatible.
- **Structured error format (QUERY-02):** Include all 12 fields from pg `DatabaseError`: `message`, `code`, `detail`, `constraint`, `table`, `schema`, `column`, `severity`, `position`, `dataType`, `hint`, `routine`. Use duck-type check (`'code' in err && 'detail' in err`) instead of `instanceof` to avoid import coupling with pg-protocol. Non-pg errors fall back to `{ message }`.
- **Query timeout implementation (QUERY-03):** Use `SET statement_timeout = <ms>` per-query (server-side cancellation). Reset to `0` in `finally` before client release. Default timeout: 0 (no limit). Per RESEARCH.md pitfall #2: ALWAYS reset to prevent timeout leakage across pooled client reuse. Also handle the `SET` query failure gracefully — if SET fails (e.g., permission denied), log a warning and proceed without timeout.
- **Timeout error detection:** When a query is cancelled by PostgreSQL due to `statement_timeout`, pg throws with `code: '57014'` (query_canceled). Detect this code and format the error message as `Query timeout after <ms>ms`.
</decisions>

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| msg.params (user input) → parameterized query | Named params pass through pg's parameter binding — safe by design |
| msg.params (user input) → Mustache template | Mustache renders raw SQL text from msg fields — UNSAFE for user-supplied data (pre-existing) |
| JavaScript catch → msg.error | Structured error fields are read-only reflections of pg error, not user-controlled |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-05 | Tampering | Mustache template injection | accept | Pre-existing from Phase 1 — Mustache interpolation of `msg` fields into SQL is unsafe for user-supplied data; documented as such; parameterized `msg.params` path (QUERY-01) is the safe alternative |
| T-02-06 | Denial of Service | statement_timeout leakage | mitigate | `statement_timeout` reset to 0 in `finally` before client release; even if SET fails, release still happens via nested try/catch; documented in RESEARCH.md pitfall #2 |
| T-02-07 | Denial of Service | Unbounded query with timeout=0 | accept | timeout=0 (unlimited) is intentional default for backward compatibility; per-node timeout field is available for users who need it; docs explain the risk |
| T-02-08 | Elevation of Privilege | SET statement_timeout permission | mitigate | If SET fails (permission denied), query proceeds without timeout and logs warning; does not block query execution |
</threat_model>
