    ```typescript
    async function listenLoop(): Promise<void> {
      let attempt = 0;

      while (!closed) {
        let client: pg.PoolClient | null = null;
        try {
          client = await node.config.pgPool.connect();
          listenerClient = client;

          // LISTEN-02: pg-format sanitized channel
          await client.query(`LISTEN ${safeChannel}`);
          attempt = 0; // D-09: reset on successful connection

          node.status({
            fill: 'green', shape: 'ring',
            text: `listening on ${config.channel}`
          });

          // LISTEN-03: notification handler with JSON parsing
          client.on('notification', (msg: pg.Notification) => {
            let payload: any = msg.payload;
            if (parseNotifyJson) {
              try { payload = JSON.parse(msg.payload || ''); } catch { /* fall through to raw string */ }
            }
            node.send({ channel: msg.channel, payload, _original: msg.payload });
          });

          // D-07: block until connection drops (error or end event)
          await new Promise<void>((_, reject) => {
            client!.on('error', reject);
            client!.on('end', reject);
          });
        } catch (err: any) {
          if (closed) break; // D-07: stop only on explicit close

          // D-08: jittered exponential backoff
          const delay = Math.min(30000, 500 * Math.pow(2, attempt)) * Math.random();
          attempt++;

          node.status({
            fill: 'yellow', shape: 'ring',
            text: `reconnecting (attempt ${attempt})`
          });

          if (client) {
            try { client.release(); } catch { /* release best-effort */ }
            listenerClient = null;
          }
          await new Promise(resolve => setTimeout(resolve, delay));
        }
      }

      // D-09: disconnected badge
      node.status({ fill: 'red', shape: 'ring', text: 'disconnected' });
    }

    listenLoop().catch((err: any) => {
      node.error(`Listener loop fatal error: ${err.message}`);
    });
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Channel name from Node-RED config | Untrusted input becomes SQL identifier in LISTEN/UNLISTEN |
| NOTIFY payload from PostgreSQL | String data from database arrives as notification payload |
| Client error/end events | Network-level events trigger reconnection logic |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-05 | Tampering | Channel name in LISTEN/UNLISTEN SQL | mitigate | `pg-format` `format('%I', channel)` escapes channel name as a PostgreSQL identifier — prevents SQL injection. Handles special characters, Unicode, quoted identifiers per PostgreSQL spec. |
| T-03-06 | Tampering | NOTIFY payload JSON injection | mitigate | Valid JSON is parsed to object; invalid JSON falls back to raw string. Downstream nodes receive a parsed object (not string) when valid — no eval(), no Function constructor. The `_original` field preserves raw payload for debugging. |
| T-03-07 | Denial of Service | Listener reconnection storm | mitigate | Jittered exponential backoff: `min(30000, 500 * 2^attempt) * Math.random()`. Full jitter prevents thundering herd. Max 30s cap prevents runaway delays. Each attempt releases the failed client. |
| T-03-08 | Denial of Service | Reconnection loop never exits | mitigate | `closed` flag set in close handler, checked at top of while loop AND after catch. If close fires during connect (connect Promise in-flight), the catch block checks `if (closed) break`. |
| T-03-09 | Information Disclosure | Channel names in logs/status text | accept | Channel name appears in node status text and log messages. This is required for observability. Channel names are developer-configured, not end-user-provided. No credentials exposed. |
</threat_model>
